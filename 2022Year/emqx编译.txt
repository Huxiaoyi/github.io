ubuntu下编译安装emqx
安装erlang24
sudo apt-get update
sudo apt-get install build-essential
sudo apt-get install libncurses5-dev m4 libssl-dev
sudo apt-get install unixodbc unixodbc-dev libc6
wget http://erlang.org/download/otp_src_24.0.tar.gz
tar -zxvf otp_src_24.0.tar.gz
cd otp_src_24.0
sudo ./configure --without-javac --prefix=/usr/local/erl 
sudo make
sudo make install


vim /etc/profile
export ERLANG_HOME=/usr/local/lib/erl/
export PATH=$PATH:$ERLANG_HOME/bin

export PATH=/usr/local/erl/bin:$PATH
source /etc/profile

#ln -s /usr/local/erl/bin/erl /usr/local/bin/erl

cd emqx

git checkout v4.3.10
